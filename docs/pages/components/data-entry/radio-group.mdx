# Radio Group

A group of radio buttons allowing single selection from a list of options.

## Import

```tsx
import { RadioGroup, RadioGroupItem } from '@riseaicloud/ui'
```

```tsx
type EdgeRuntime = 'none' | 'openyurt' | 'kubeedge'

function EdgeRuntimeStep() {
  const [edgeRuntime, setEdgeRuntime] = useState<EdgeRuntime>('openyurt')
  return (
    <RadioGroup
      value={edgeRuntime}
      onValueChange={(v) => setEdgeRuntime(v as EdgeRuntime)}
      className="space-y-3"
    >
      {[
        { value: 'none', label: '不安装边缘组件', desc: '作为普通 Kubernetes 集群使用' },
        { value: 'openyurt', label: 'OpenYurt', desc: '阿里巴巴开源的云边协同平台', recommended: true },
        { value: 'kubeedge', label: 'KubeEdge', desc: 'CNCF 边缘计算项目' },
      ].map(({ value, label, desc, recommended }) => (
        <label key={value} className={cn(
          "flex items-center gap-3 cursor-pointer rounded-lg border p-3 transition-colors",
          edgeRuntime === value ? "border-blue-300 bg-blue-50" : "border-gray-200 hover:border-gray-300"
        )}>
          <RadioGroupItem value={value} id={`rt-${value}`} />
          <div className="flex-1">
            <div className="flex items-center gap-2">
              <Label htmlFor={`rt-${value}`} className="font-medium cursor-pointer">
                {label}
              </Label>
              {recommended && (
                <span className="text-xs px-2 py-0.5 bg-green-100 text-green-700 rounded">推荐</span>
              )}
            </div>
            <div className="text-sm text-gray-500 mt-0.5">{desc}</div>
          </div>
        </label>
      ))}
    </RadioGroup>
  )
}
```

## Controlled

```tsx
import { useState } from 'react'

function ControlledRadioGroup() {
  const [value, setValue] = useState('manual')
  return (
    <RadioGroup value={value} onValueChange={setValue}>
      <div className="flex items-center gap-2">
        <RadioGroupItem value="manual" id="r-manual" />
        <Label htmlFor="r-manual">手动导入</Label>
      </div>
      <div className="flex items-center gap-2">
        <RadioGroupItem value="auto" id="r-auto" />
        <Label htmlFor="r-auto">自动创建</Label>
      </div>
    </RadioGroup>
  )
}
```

## API

### RadioGroup Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | — | Controlled selected value |
| `defaultValue` | `string` | — | Initial value (uncontrolled) |
| `onValueChange` | `(value: string) => void` | — | Callback when selection changes |
| `className` | `string` | — | Additional CSS classes (default: `space-y-2`) |

### RadioGroupItem Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `value` | `string` | — | The value this radio button represents |
| `disabled` | `boolean` | `false` | Disabled state |
| `className` | `string` | — | Additional CSS classes |
