### Usage

```tsx
import { useState } from 'react'
import InviteMemberDialog from '@riseaicloud/components'

function ClusterMembers({ clusterId, members }) {
  const [open, setOpen] = useState(false)

  const handleInvite = async (usernames: string[], role: string) => {
    await Promise.all(
      usernames.map((username) =>
        addClusterMember(clusterId, { username, role })
      )
    )
  }

  return (
    <>
      <Button onClick={() => setOpen(true)}>邀请成员</Button>

      <InviteMemberDialog
        isOpen={open}
        onClose={() => setOpen(false)}
        onInvite={handleInvite}
        clusterId={clusterId}
        existingMembers={members.map((m) => m.username)}
        availableRoles={[
          { value: 'viewer', label: '查看者', description: '只读访问' },
          { value: 'editor', label: '编辑者', description: '可修改资源' },
          { value: 'admin', label: '管理员', description: '完全控制' },
        ]}
        type="cluster"
      />
    </>
  )
}
```

### Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `isOpen` | `boolean` | — | Whether the dialog is open |
| `onClose` | `() => void` | — | Close callback |
| `onInvite` | `(usernames: string[], role: string) => Promise<void>` | — | Async invite callback |
| `clusterId` | `string` | — | Cluster identifier (for fetching users) |
| `existingMembers` | `string[]` | — | Usernames already in the team (filtered from selection) |
| `availableRoles` | `Array<{ value: string; label: string; description: string }>` | — | Roles to offer in the dropdown |
| `type` | `'cluster' \| 'nodegroup' \| 'namespace' \| 'workspace'` | `'cluster'` | Context label displayed in the dialog |

### Behavior

**User Loading**
- Fetches platform users via `listUsers()` on open; supports pagination (`pageSize: 20`)
- Infinite scroll: loads more when list container is scrolled near the bottom
- Debounced search (500ms) filters by username or display name

**Selection**
- Users already in `existingMembers` are shown disabled
- Selected users appear as removable chips
- Each selected user gets an individual role dropdown (defaults to first available role)
- Invite button disabled if no users are selected

**Layout**
- Left panel: searchable user list with checkboxes
- Right panel: selected users with per-user role assignment
- Fixed `sm:max-w-[640px]` dialog width
